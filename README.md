```C#
using System; // Подключаем пространство имен System для использования встроенных функций, таких как Console.

class Program // Объявляем класс с именем Program — точка входа для выполнения программы.
{
    static void Main(string[] args) // Главный метод, с которого начинается выполнение программы.
    {
        // Сообщение пользователю о вводе номера дня недели.
        Console.WriteLine("Введите номер дня недели (1-7):");

        // Читаем ввод пользователя, преобразуем в целое число и сохраняем в переменную day.
        int day = int.Parse(Console.ReadLine());

        // Конструкция switch проверяет значение переменной day.
        switch (day)
        {
            // Если day равно 1, выполняется следующий блок кода:
            case 1:
                Console.WriteLine("Понедельник — рабочий день."); // Выводим сообщение для понедельника.
                break; // Завершаем выполнение текущего case, чтобы не выполнять другие case.
            
            // Если day равно 2, выполняется следующий блок кода:
            case 2:
                Console.WriteLine("Вторник — рабочий день."); // Выводим сообщение для вторника.
                break;

            // Если day равно 3, выполняется следующий блок кода:
            case 3:
                Console.WriteLine("Среда — рабочий день."); // Выводим сообщение для среды.
                break;

            // Если day равно 4, выполняется следующий блок кода:
            case 4:
                Console.WriteLine("Четверг — рабочий день."); // Выводим сообщение для четверга.
                break;

            // Если day равно 5, выполняется следующий блок кода:
            case 5:
                Console.WriteLine("Пятница — рабочий день, но скоро выходные!"); // Выводим сообщение для пятницы.
                break;

            // Если day равно 6 или 7, выполняется следующий блок кода:
            case 6: // День = суббота.
            case 7: // День = воскресенье.
                Console.WriteLine("Выходной день! Наслаждайтесь отдыхом."); // Выводим сообщение для выходного.
                break;

            // Если значение переменной day не соответствует ни одному из case:
            default:
                Console.WriteLine("Ошибка: введите число от 1 до 7."); // Сообщаем об ошибке ввода.
                break;
        }
    }
}

```
